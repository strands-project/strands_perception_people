cmake_minimum_required(VERSION 2.8.3)
project(human_trajectory)

find_package(catkin REQUIRED COMPONENTS
  tf
  message_filters
  geometry_msgs
  rospy
  std_msgs
  message_generation
  strands_navigation_msgs
  mongodb_store
  bayes_people_tracker
  topological_logging_manager
)

 catkin_python_setup()

################################################
## Declare ROS messages, services and actions ##
################################################
add_message_files(
  FILES
  Trajectory.msg
  Trajectories.msg
)

generate_messages(
  DEPENDENCIES
  geometry_msgs std_msgs
)

###################################
## catkin specific configuration ##
###################################
catkin_package(
  CATKIN_DEPENDS geometry_msgs rospy std_msgs message_runtime mongodb_store strands_navigation_msgs bayes_people_tracker topological_logging_manager tf message_filters
)

###########
## Build ##
###########

#include_directories(
  #${catkin_INCLUDE_DIRS}
#)

#############
## Install ##
#############

 install(PROGRAMS
   scripts/trajectory_publisher.py
   scripts/traj_visualisation.py
   DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
 )

install(DIRECTORY launch
    DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
